import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from io import StringIO

# Sample dataset as provided
data_csv = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,IT,Middle East,90.66,12,4.3
EMP002,IT,Africa,94.11,7,3.9
EMP003,R&D,Africa,72.75,4,4.0
EMP004,Marketing,Europe,90.38,10,3.2
EMP005,Marketing,Latin America,80.06,7,3.2
EMP006,Sales,North America,88.45,8,4.1
EMP007,Finance,Asia,85.23,6,3.8
EMP008,Marketing,Asia,92.17,9,4.5
EMP009,IT,Europe,87.92,11,3.7
EMP010,R&D,North America,79.34,5,4.2
EMP011,Sales,Middle East,83.67,7,3.6
EMP012,Finance,Africa,90.12,12,4.0
EMP013,Marketing,Middle East,86.78,8,3.9
EMP014,IT,Latin America,91.56,10,4.3
EMP015,R&D,Europe,77.89,4,3.5
EMP016,Sales,Asia,89.23,9,4.1
EMP017,Finance,North America,84.56,6,3.8
EMP018,Marketing,Africa,88.91,7,4.0
EMP019,IT,Asia,93.45,13,4.4
EMP020,R&D,Middle East,81.23,5,3.7"""

# Load the data
print("Loading employee performance data...")
df = pd.read_csv(StringIO(data_csv))

print(f"Dataset loaded successfully with {len(df)} employees")
print(f"Departments: {df['department'].unique()}")
print(f"Regions: {df['region'].unique()}")
print()

# Calculate frequency count for Marketing department
marketing_count = df[df['department'] == 'Marketing'].shape[0]
print(f"FREQUENCY COUNT FOR MARKETING DEPARTMENT: {marketing_count}")
print()

# Display basic statistics
print("=== DEPARTMENT DISTRIBUTION ===")
dept_counts = df['department'].value_counts()
for dept, count in dept_counts.items():
    print(f"{dept}: {count} employees")
print()

# Create visualization
plt.style.use('seaborn-v0_8')
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))

# Histogram of department distribution
dept_counts.plot(kind='bar', ax=ax1, color=['#3498db', '#e74c3c', '#2ecc71', '#f39c12', '#9b59b6'])
ax1.set_title('Employee Distribution by Department', fontsize=16, fontweight='bold')
ax1.set_xlabel('Department', fontsize=12)
ax1.set_ylabel('Number of Employees', fontsize=12)
ax1.tick_params(axis='x', rotation=45)
ax1.grid(axis='y', alpha=0.3)

# Add value labels on bars
for i, v in enumerate(dept_counts.values):
    ax1.text(i, v + 0.1, str(v), ha='center', va='bottom', fontweight='bold')

# Performance score distribution by department
df.boxplot(column='performance_score', by='department', ax=ax2)
ax2.set_title('Performance Score Distribution by Department', fontsize=16, fontweight='bold')
ax2.set_xlabel('Department', fontsize=12)
ax2.set_ylabel('Performance Score', fontsize=12)
plt.suptitle('')  # Remove automatic title

plt.tight_layout()
plt.show()

# Additional analysis
print("=== PERFORMANCE INSIGHTS ===")
avg_performance = df.groupby('department')['performance_score'].agg(['mean', 'std']).round(2)
print("Average Performance Score by Department:")
print(avg_performance)
print()

print("=== REGIONAL INSIGHTS ===")
regional_dist = df.groupby(['region', 'department']).size().unstack(fill_value=0)
print("Employee Distribution by Region and Department:")
print(regional_dist)
print()

# Marketing department specific analysis
marketing_dept = df[df['department'] == 'Marketing']
print("=== MARKETING DEPARTMENT ANALYSIS ===")
print(f"Total Marketing employees: {len(marketing_dept)}")
print(f"Average performance score: {marketing_dept['performance_score'].mean():.2f}")
print(f"Average years of experience: {marketing_dept['years_experience'].mean():.2f}")
print(f"Average satisfaction rating: {marketing_dept['satisfaction_rating'].mean():.2f}")
print()

print("Marketing employees by region:")
marketing_by_region = marketing_dept['region'].value_counts()
for region, count in marketing_by_region.items():
    print(f"  {region}: {count} employees")

print("\n=== ANALYSIS COMPLETE ===")
print("This analysis was prepared for strategic workforce planning.")
print("Contact: 23f1002492@ds.study.iitm.ac.in")
